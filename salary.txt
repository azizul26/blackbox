			function calculateSalary() 
			{
				
			
			var stf = document.getElementById("position").value;
			var wh = document.getElementById("HoursWorked").value;
			
				if (stf == "Staff")
				{   
				    if(wh > 40)
					{
					var ot = wh - 40;
					var basicHours = wh - ot;
					}
					else
					{
					var ot = 0;
					var basicHours = wh - ot;
					}
					
					
					var calcOt = 1.5 * ot;
					var basicPay = basicHours * 6.5;
					
					var grossPay = calcOt + basicPay;
					var taxes = grossPay * 0.06;
					
					var netPay = grossPay - taxes;
					
					outputGrossPay.value = grossPay;
					outputTaxes.value = taxes;
					outputNett.value = netPay;
				}
				else if (stf == "Super")
				{
					if(wh > 40)
					{
					var ot = wh - 40;
					var basicHours = wh - ot;
					}
					else
					{
					var ot = 0;
					var basicHours = wh - ot;
					}
					
					var calcOt = 1.5 * ot;
					var basicPay = basicHours * 7.5;
					
					var grossPay = calcOt + basicPay;
					
					if(document.getElementById("TaxCheck").checked == true)
					{var taxes = 0; }
					else
					{var taxes = 0.06 * grossPay;}
					
					
					
					var netPay = grossPay - taxes;
					
					outputGrossPay.value = grossPay;
					outputTaxes.value = taxes;
					outputNett.value = netPay;
				}
				else 
				{
					if(wh > 40)
					{
					var ot = wh - 40;
					var basicHours = wh - ot;
					}
					else
					{
					var ot = 0;
					var basicHours = wh - ot;
					}
					var calcOt = 1.5 * ot;
					var basicPay = basicHours * 15;
					
					var grossPay = calcOt + basicPay;
					if(document.getElementById("TaxCheck").checked == true)
					{var taxes = 0; }
					else
					{var taxes = 0.06 * grossPay;}
					var netPay = grossPay - taxes;
					
					outputGrossPay.value = grossPay;
					outputTaxes.value = taxes;
					outputNett.value = netPay;
				}
				//table2.hidden = false;
			}

			function checkPosition()
			{
				var x = document.getElementById("position").value;
				
				if (x == "Super")
				{
				    document.getElementById("TaxCheck").checked = false;
					document.getElementById("TaxCheck").disabled = false;
				}
				else if (x == "Manager")
				{   
				    document.getElementById("TaxCheck").checked = false;
					document.getElementById("TaxCheck").disabled = false;
				}
				else if (x == "Staff")
				{
				    document.getElementById("TaxCheck").checked = false;
					document.getElementById("TaxCheck").disabled = true;
				}
			}

			function checkField()
			{       
			        if(document.getElementById("HoursWorked").value == "")
					{
					document.getElementById('btnCalc').disabled = true;
					}
					else{
					document.getElementById('btnCalc').disabled = false;
					}
			}
			
		function isNumber(evt) 
		{
         evt = (evt) ? evt : window.event;
         var charCode = (evt.which) ? evt.which : evt.keyCode;
         if (charCode > 31 && (charCode < 48 || charCode > 57)) 
		 {
            return false;
         }
         return true;
        }
